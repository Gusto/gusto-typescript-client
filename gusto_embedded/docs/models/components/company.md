# Company

The representation of a company in Gusto.

## Example Usage

```typescript
import { Company } from "@gusto/embedded-api/models/components/company.js";

let value: Company = {
  uuid: "39211506-a611-442f-a2c9-98f5c86f916c",
  compensations: {
    hourly: [
      {
        name: "Overtime",
        multiple: 1.5,
      },
    ],
    fixed: [
      {
        name: "Bonus",
      },
    ],
    paidTimeOff: [
      {
        name: "Vacation Hours",
      },
    ],
  },
};
```
