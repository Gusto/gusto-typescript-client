# Employee

The representation of an employee in Gusto.

## Example Usage

```typescript
import { Employee } from "openapi/models/components";

let value: Employee = {
  uuid: "3917557b-1eb2-4f8f-8384-49ecb8e7dd3c",
  firstName: "Enoch",
  lastName: "Lakin",
  eligiblePaidTimeOff: [
    {
      accrualUnit: "Hour",
      accrualMethod: "unlimited",
      accrualPeriod: "Year",
    },
  ],
};
```
