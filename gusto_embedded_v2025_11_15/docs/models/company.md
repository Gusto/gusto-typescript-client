# Company

The representation of a company in Gusto.

## Example Usage

```typescript
import { Company } from "openapi/models";

let value: Company = {
  uuid: "9ca13f72-0a2d-4f1c-bd18-fa78e3dc0991",
  compensations: {
    hourly: [
      {
        name: "Overtime",
        multiple: 1.5,
      },
    ],
    fixed: [
      {
        name: "Bonus",
      },
    ],
    paidTimeOff: [
      {
        name: "Vacation Hours",
      },
    ],
  },
};
```
